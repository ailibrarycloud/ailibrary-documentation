### Step 5: Chat

The agent chat endpoint either streams text responses (default behaviour) or returns a json response (please see the Agent section for details). The ailibrary Python package provides client.agent.chat() (for either json or text response format) and client.agent.chat_stream() so you can interact with the agent however you prefer.

<CodeGroup title="Chat">

```python
namespace = "sales-assistant-abc123"
session_id = "niusjndckj23"
messages = [
    { "role": "assistant", "content": "how are you?" },
    { "role": "user", "content": "doing good. thank you." }
]

# ================== NON STREAMING OPTION ==================
response = client.agent.chat(
    namespace=namespace,
    messages=messages,
    session_id=session_id
    response_format="text" # or "json"
)
print(response)

# ==================== STREAMING OPTION ====================
result = ""
stream = client.agent.chat_stream(
    namespace=name,

```

```javascript
let session_id = "niusjndckj23"
let messages = [{ role: "assistant", content: "how are you?" }];
async function makeRequest(prompt) {
    messages.push({
        id: Date.now(),
        role: "user",
        content: prompt,
    });

    try {
        const response = await fetch(`${domain}/v1/agent/${namespace}/chat`, {
            method: "POST",
            headers: {
                "X-Library-Key": key,
                "Content-Type": "application/json",
            },
            body: JSON.stringify({
                stream: "true",
                session_id: session_id,
                messages: messages,
            }),
    });

    if (!response.body) {
        throw new Error("ReadableStream not supported in this browser.");
    }

    const reader = response.body.getReader();

    const decoder = new TextDecoder();
    let result = "";

    const processChunk = ({ done, value }) => {
        if (done) {
        setMessageLoader(false);
        console.log("Stream complete", latency);
        messages.push({
            id: Date.now(),
            role: "assistant",
            content: result,
            latency: latency,
            knowledge: knowledge.length > 0 ? knowledge : null,
        });
        setMessages(messages);
        setResponse("");
        return;
        }

        const chunk = decoder.decode(value, { stream: true });
        const jsonStrings = chunk.split("\n");

        jsonStrings.forEach((jsonString) => {
        if (jsonString.trim() !== "") {
            try {
            const parsed = JSON.parse(jsonString);
            console.log("parsed", parsed.content);
            if (parsed.object === "chat.completion.chunk") {
                result += parsed.content;
                setResponse((prevResponse) => prevResponse + parsed.content);
            }
            if (parsed.object === "chat.completion.latency") {
                console.log("latency", parsed.content);
                latency = parsed.content;
            }
            if (parsed.object === "chat.completion.form") {
                console.log("form", parsed.content);
                // setForm(parsed.content);
            }
            if (parsed.object === "chat.completion.knowledge") {
                console.log("knowledge", parsed.content);
                knowledge = parsed.content;
            }
            } catch (error) {
                console.error("Error parsing chunk:", error);
            }
        }
        });

        // Read the next chunk
        return reader.read().then(processChunk);
    };

    // Start reading the stream
    reader.read().then(processChunk);
} catch (error) {
    console.log(error);
}
}
```
